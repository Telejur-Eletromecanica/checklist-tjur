function enviarWhatsApp() { const salvo = JSON.parse(localStorage.getItem("tjur_checklist_ativo")); const kmf = document.getElementById("kmf").value; if(!kmf) { alert("Informe o KM Final!"); return; } const total = (parseFloat(kmf) - parseFloat(salvo.kmi)).toFixed(1); let proxOleo = salvo.veiculo === "SKY MUNCK" ? (salvo.oleoData ? salvo.oleoData.split('-').reverse().join('/') : "NÃO INFORMADO") : (salvo.oleoKM ? salvo.oleoKM + " KM" : "NÃO INFORMADO"); // FORMATAÇÃO PARA WHATSAPP // Título com sublinhado duplo (simulado com sinal de igual) // Duas quebras de linha após o título // Demais linhas com sublinhado simples (simulado com traços) e uma quebra entre elas const mensagem = `*CHECKLIST VEÍCULOS - TELEJUR*\n` + `==========================\n\n\n` + // Sublinhado duplo + 2 quebras `*DATA:* ${salvo.data}\n` + `--------------------------\n\n` + // Sublinhado simples + 1 quebra `*ID:* ${salvo.num}\n` + `--------------------------\n\n` + `*CONDUTOR:* ${salvo.nome}\n` + `--------------------------\n\n` + `*TAG:* ${salvo.tag}\n` + `--------------------------\n\n` + `*VEÍCULO:* ${salvo.veiculo}\n` + `--------------------------\n\n` + `*PLACA:* ${salvo.placa}\n` + `--------------------------\n\n` + `*KM INICIAL:* ${salvo.kmi}\n` + `--------------------------\n\n` + `*KM FINAL:* ${kmf}\n` + `--------------------------\n\n` + `*TOTAL:* ${total} KM\n` + `--------------------------\n\n` + `*PRÓX. ÓLEO:* ${proxOleo}\n` + `--------------------------\n\n` + `*OBS:* ${document.getElementById("obs").value || "SEM OBSERVAÇÃO"}\n` + `--------------------------`; localStorage.removeItem("tjur_checklist_ativo"); window.location.href = "whatsapp://send?text=" + encodeURIComponent(mensagem); }
