async function verificarStatus() { const tag = document.getElementById('tag_select').value; const idCampo = document.getElementById('id_condutor'); const btn = document.getElementById('btn_enviar'); idCampo.value = "VERIFICANDO..."; try { const r = await fetch(`${scriptURL}?tag=${encodeURIComponent(tag)}&_=${new Date().getTime()}`); const d = await r.json(); if (d.OCUPADO) { alert(`ATENÇÃO: A ${tag} já está em uso por ${d.MOTORISTA}. Finalize o trajeto anterior primeiro.`); idCampo.value = "TAG EM USO"; idCampo.style.color = "red"; btn.disabled = true; } else { // Aqui ele preenche automaticamente Condutor 1, 2, ou 3... idCampo.value = d.SUGESTAOID; idCampo.style.color = "blue"; btn.disabled = false; } } catch (e) { idCampo.value = "ERRO DE CONEXÃO"; } }
