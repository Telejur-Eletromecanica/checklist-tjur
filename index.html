async function validarVinculo() { const tag = document.getElementById("tag").value; const idCond = document.getElementById("numCondutor").value; // Ex: "CONDUTOR 1" const nomeAtual = document.getElementById("nomeCondutor").value; const btn = document.getElementById("btnAcao"); const msg = document.getElementById("msgErroVinc"); // Só valida se tiver TAG e ID selecionados if (!tag || !idCond) { msg.style.display = "none"; return; } msg.style.display = "block"; msg.style.background = "#fff3cd"; msg.innerHTML = "Verificando disponibilidade do ID para esta TAG..."; btn.disabled = true; try { // A URL envia a TAG e o ID para o seu script (acao=verificar) const urlBusca = `${SCRIPT_URL}?acao=verificar&tag=${encodeURIComponent(tag)}&id=${encodeURIComponent(idCond)}&data=${hoje}`; const response = await fetch(urlBusca); const ocupante = await response.text(); // LÓGICA: Se o script retornar um nome que NÃO é o meu, e não estiver "livre", bloqueia. if (ocupante !== "livre" && ocupante !== "" && ocupante !== nomeAtual) { msg.style.background = "#f8d7da"; msg.style.color = "#721c24"; msg.innerHTML = `🚫 **BLOQUEADO:** O **${idCond}** na **${tag}** já está ocupado por **${ocupante}**.`; btn.disabled = true; } else { // Está livre para este ID ou o ocupante sou eu mesmo msg.style.display = "none"; btn.disabled = false; } } catch (e) { // Em caso de falha na internet, libera para não travar a operação console.error("Erro na validação:", e); msg.style.display = "none"; btn.disabled = false; } }
